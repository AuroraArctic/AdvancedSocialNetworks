sample_matrix<-matrix(NA,1000,1) # Initializing the matrix

for (k in c(1:1000))
{
  PFA.P_PERM<-sample(PFA.P) # Permutation of Priors
  # Compute correlation between degree and Priors permuted
  sample_matrix[k,1]<-cor(PFMn.Deg,PFA.P_PERM) 
}
